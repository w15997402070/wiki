# 启动

* 启动命令

  ```bash
  # 启动hadoop
  [root@localhost hadoop-3.2.1]# sbin/start-all.sh
  # 关闭hadoop
  [root@localhost hadoop-3.2.1]# sbin/stop-all.sh
  
  # 启动hbase
  [root@localhost hbase-2.2.4]# ./bin/start-hbase.sh
  # 关闭hbase
  [root@localhost hbase-2.2.4]# ./bin/stop-hbase.sh
  # 启动hbase shell
  [root@localhost hbase-2.2.4]# ./bin/hbase shell
  ```

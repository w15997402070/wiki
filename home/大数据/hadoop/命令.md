# 命令

```shell
# 删除
hdfs dfs -rm -r -f /hbase

# 查看
hdfs dfs -ls /
#创建文件夹,单层创建不能循环
hdfs dfs -mkdir /wordcount/output
#循环创建
hdfs dfs -mkdir -p /wordcount/output



bin/zkServer.sh start
bin/zkServer.sh status

hdfs jar bdata-1.0.jar com.wang.hadoop.WordCountDriver /wordcount/input /wordcount/output

```

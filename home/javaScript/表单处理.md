1.checkbox处理
```js
    //checkbox获取值处理
    $(':checkbox').each(function () {
        var $this = $(this);
        var name = $this.data('name');
        if(!name) return true;
        extras[name] = $this.prop('checked')?1:0;
    });
 ```
